#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstring>
using namespace std;

int n, m;
int map[110];
int dist[110];
queue<int> q;

void input() {
	scanf("%d%d",&n,&m);
	for (int i = 0; i < n + m; i++) {
		int s, e;
		scanf("%d %d",&s, &e);
		map[s] = e;
	}
}


int main() {
	//freopen("input.txt", "r", stdin);
	input();

	memset(dist, -1, sizeof(dist));
	q.push(1);
	dist[1] = 0;
	while (true)
	{
		if (q.empty()) break;
		int temp = q.front();
		q.pop();

		for (int dice = 1; dice <= 6; dice++)
		{
			int nextNode = temp + dice;
			if (nextNode > 100) continue;

			if (map[nextNode] != 0) {
				nextNode = map[nextNode];
			}

			if (dist[nextNode] == -1) {
				dist[nextNode] = dist[temp] + 1;
				q.push(nextNode);
			}
		}
	}

	printf("%d",dist[100]);

	return 0;
}
