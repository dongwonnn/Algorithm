function solution(s)
{
    function run(arr){
    const mid = Math.floor(arr.length/2);

    for(let i = 0 ; i < mid; i++){
        if(arr[i] !== arr[arr.length-1-i]) return false;
    }

    return true;
}

    for(let i = s.length; i>=1; i--){
        for(let j = 0; j<=s.length -i; j++){
            const isCheck = run(s.slice(j, i+j));
            if(isCheck) return i;
        }
    }

    return 1;
}
