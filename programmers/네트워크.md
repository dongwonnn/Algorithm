function solution(n, computers) {
  let ans = 0;
  const check = Array(n).fill(false);

  function go(idx) {
    check[idx] = true;

    for (let i = 0; i < computers.length; i++) {
      if (computers[idx][i] === 1 && !check[i]) go(i);
    }
  }

  for (let i = 0; i < computers.length; i++) {
    if (!check[i]) {
      go(i);
      ans++;
    }
  }

  return ans;
}
