function solution(begin, target, words) {
  begin = begin.split('');
  target = target.split('');
  const len = begin.length;

  let check = new Array(words.length + 1).fill(false);
  let ans = [];

  function DFS(str, depth) {
    if (str.join('') === target.join('')) {
      ans.push(depth);
    }

    for (let i = 0; i < words.length; i++) {
      if (check[i]) continue;
      let cnt = 0;

      let word = words[i].split('');
      for (let i = 0; i < len; i++) {
        if (str[i] === word[i]) cnt++;
      }

      if (cnt === len - 1) {
        check[i] = true;
        DFS(word, depth + 1);
        check[i] = false;
      }
    }
  }

  DFS(begin, 0);

  //console.log(Math.min.apply(null, ans));

  if (ans.length === 0) return 0;
  else return Math.min.apply(null, ans);
}
