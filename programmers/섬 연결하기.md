function solution(n, costs) {
  let ans = 0;
  let cnt = 0;
  let bridge_check = [];
  costs.sort((a, b) => a[2] - b[2]);

  let check = [];
  check[costs[0][0]] = true;
  check[costs[0][1]] = true;
  bridge_check[0] = true;
  ans += costs[0][2];
  cnt = 2;

  while (cnt < n) {
    for (let i = 0; i < costs.length; i++) {
      const [s, e, cost] = costs[i];

      if (
        (!bridge_check[i] && !check[s] && check[e]) ||
        (check[s] && !check[e])
      ) {
        check[s] = true;
        check[e] = true;
        bridge_check[i] = true;
        cnt++;
        ans += cost;
        break;
      }
    }
  }

  return ans;
}
