function solution(tickets) {
  console.log(tickets);

  let check = new Array(tickets.length).fill(false);
  let ans = [];

  function DFS(idx, depth, root) {
    const next = tickets[idx][1];
    if (depth === tickets.length - 1) {
      root.push(next);

      if (ans.length === 0) {
        ans = root.slice();
      } else if (ans > root) {
        ans = root.slice();
      }
      root.pop(next);
      return;
    }

    // SFO

    for (let i = 0; i < tickets.length; i++) {
      if (check[i]) continue;
      if (tickets[i][0] === next) {
        check[i] = true;
        root.push(next);
        DFS(i, depth + 1, root);
        check[i] = false;
        root.pop();
      }
    }
  }

  for (let i = 0; i < tickets.length; i++) {
    if (tickets[i][0] === 'ICN') {
      check[i] = true;
      let root = ['ICN'];
      DFS(i, 0, root);
      check[i] = false;
      root.pop();
    }
  }

  return ans;
}
