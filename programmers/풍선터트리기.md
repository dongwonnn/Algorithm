function solution(a) {
//   const minValue = Math.min.apply(null, a);

//   console.log(minValue);

//   a = a.map((v) => v - minValue);
//   console.log(a);


  let left = a[0];
  let right = new Array(a.length);
  let min = 1000000010;

  for(let i = a.length-1 ; i > 0; i--){
    if(a[i]< min) min = a[i];
    right[i] = min;
  }

  let ans = a.length < 2? 1:2;

  for(let i = 1 ; i< a.length-1; i++){
    if(left > a[i]) left = a[i];
    if(left < a[i] && right[i] < a[i] ) continue;
    ans++;
  }


  return ans;
}
