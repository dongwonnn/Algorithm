function solution(n, s, a, b, fares) {
  let dist = Array.from(Array(n + 1), () => Array(n + 1).fill(Infinity));

  for (let i = 1; i <= n; i++) {
    dist[i][i] = 0;
  }

  for (let i = 0; i < fares.length; i++) {
    const start = fares[i][0];
    const end = fares[i][1];
    const dist = fares[i][2];

    dist[start][end] = dist;
    dist[end][start] = dist;
  }

  for (let i = 1; i <= n; i++) {
    for (let j = 1; j <= n; j++) {
      for (let k = 1; k <= n; k++) {
        if (dist[j][i] != Infinity && dist[i][k] != Infinity) {
          dist[j][k] = Math.min(dist[j][k], dist[j][i] + dist[i][k]);
        }
      }
    }
  }

  let answer = Infinity;

  for (let i = 1; i <= n; i++)
    answer = Math.min(answer, dist[s][i] + dist[i][a] + dist[i][b]);

  return answer;
}
