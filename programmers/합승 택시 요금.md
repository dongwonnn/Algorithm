function solution(n, s, a, b, fares) {
  let adj = Array.from(Array(n + 1), () => Array(n + 1).fill(Infinity));

  for (let i = 1; i <= n; i++) {
    adj[i][i] = 0;
  }

  for (let i = 0; i < fares.length; i++) {
    const start = fares[i][0];
    const end = fares[i][1];
    const adj = fares[i][2];

    adj[start][end] = adj;
    adj[end][start] = adj;
  }

  for (let i = 1; i <= n; i++) {
    for (let j = 1; j <= n; j++) {
      for (let k = 1; k <= n; k++) {
        if (adj[j][i] != Infinity && adj[i][k] != Infinity) {
          adj[j][k] = Math.min(adj[j][k], adj[j][i] + adj[i][k]);
        }
      }
    }
  }

  let answer = Infinity;

  for (let i = 1; i <= n; i++)
    answer = Math.min(answer, adj[s][i] + adj[i][a] + adj[i][b]);

  return answer;
}
